DIPLOMACY_JOIN_ALLY_ENABLE_TRIGGER = {
	if = {
		limit = {
			AND = {
				tag = SPR
				has_idea = SPA_no_navy
			}
		}
		custom_trigger_tooltip = {
			tooltip = SPAIN_ALLIES_NEUTRAL
			always = no
		}
	}
	
	else = {
		always = yes
	}
}
